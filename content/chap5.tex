\section{Discretization Methods}

\subsection{Euler And Milstein Schemes}

\subsubsection*{Euler scheme}

For the SDE $dX(t)=a(X(t))\,dt+b(X(t))\,dW(t)$, with $X\in\mathbb{R}^d$ and $W\in\mathbb{R}^m$, the Euler approximation on $t_i=ih$ is

$\hat X(i+1)=\hat X(i)+a(\hat X(i))h+b(\hat X(i))\sqrt h\,Z_{i+1}$,

where $Z_i$ are independent $m$-dimensional standard normals. Euler freezes the drift and diffusion integrands on each interval. The drift is order $h$, while the diffusion increment is order $\sqrt h$, so a first refinement focuses on the diffusion term.

\subsubsection*{Scalar Milstein refinement}

In the scalar case, Ito's formula gives

$db(X(t))=\left[b'(X(t))a(X(t))+\frac12 b''(X(t))b^2(X(t))\right]dt
\\+b'(X(t))b(X(t))\,dW(t)$.

Dropping the higher-order drift part,

$b(X(u))\approx b(X(t))+b'(X(t))b(X(t))[W(u)-W(t)]$.

With $\Delta W=W(t+h)-W(t)$,

$\int_t^{t+h}b(X(u))\,dW(u)\approx b(X(t))\Delta W+
\\b'(X(t))b(X(t))\int_t^{t+h}[W(u)-W(t)]\,dW(u)$.

The remaining integral is

$\int_t^{t+h}[W(u)-W(t)]\,dW(u)=\frac12(\Delta W)^2-\frac12h$.

Thus the Milstein one-step approximation is

$X(t+h)\approx X(t)+a(X(t))h+b(X(t))\Delta W+\frac12b'(X(t))b(X(t))\{(\Delta W)^2-h\}$.

Equivalently, with $\Delta W=\sqrt h\,Z_{i+1}$,

$\hat X(i+1)=\hat X(i)+a(\hat X(i))h+b(\hat X(i))\sqrt h\,Z_{i+1}+\frac12 b'(\hat X(i))b(\hat X(i))h(Z_{i+1}^2-1)$.

Conditional on $\hat X(i)$, the extra term has mean $0$ and is uncorrelated with the Euler diffusion term since $Z^2-1$ and $Z$ are uncorrelated.

\subsubsection*{GBM comparison}

For $dX_t=rX_t\,dt+\sigma X_t\,dW_t$, Euler is

$\hat X_{t_{i+1}}=\hat X_{t_i}+r\hat X_{t_i}h+\sigma\hat X_{t_i}\sqrt h\,Z_{i+1}$.

Milstein is

$\hat X_{t_{i+1}}=\hat X_{t_i}+r\hat X_{t_i}h+\sigma\hat X_{t_i}\sqrt h\,Z_{i+1}+\frac12\sigma^2\hat X_{t_i}h(Z_{i+1}^2-1)$.

The exact benchmark is

$X_{t_{i+1}}=X_{t_i}\exp\{(r-\frac12\sigma^2)h+\sigma\sqrt h\,Z_{i+1}\}$.

In the lecture's numerical comparison, Euler strong error is order $\sqrt h$, while Milstein error is order $h$.

\subsubsection*{Multidimensional difficulty}

For components $X_i$, $W_j$, $a_i$, and $b_{ij}$,

$X_i(t+h)=X_i(t)+\int_t^{t+h}a_i(X(u))\,du+\sum_{j=1}^m\int_t^{t+h}b_{ij}(X(u))\,dW_j(u)$.

The diffusion refinement contains

$\int_t^{t+h} b_{ij}(X(u))\,dW_j(u)\approx b_{ij}(X(t))\Delta W_j
\\+\sum_{\ell=1}^d\sum_{k=1}^m\frac{\partial b_{ij}}{\partial x_\ell}(X(t))b_{\ell k}(X(t))I_{kj}$,

where

$I_{kj}=\int_t^{t+h}[W_k(u)-W_k(t)]\,dW_j(u)$.

For $k=j$, $I_{jj}=\frac12(\Delta W_j)^2-\frac12h$. For $k\ne j$, the mixed integrals have no comparable simple formula; their differences are Levy area terms, and sampling them is difficult.

\subsection{Lamperti Transform}

\subsubsection*{Definition}

For $dX_t=b(X_t)\,dt+\sigma(X_t)\,dW_t$ with $\sigma(x)>0$, define

$Y_t=F(X_t),\qquad F(x)=\int\frac{1}{\sigma(x)}\,dx$.

Then $F$ is increasing and $X_t=F^{-1}(Y_t)$. Since

$F'(x)=1/\sigma(x),\qquad F''(x)=-\sigma'(x)/\sigma^2(x)$,

Ito's formula gives

$dY_t=\left[\frac{b(X_t)}{\sigma(X_t)}-\frac12\sigma'(X_t)\right]dt+dW_t$.

Writing $X_t=F^{-1}(Y_t)$,

$dY_t=a(Y_t)\,dt+dW_t,\qquad a(y)=\frac{b(F^{-1}(y))}{\sigma(F^{-1}(y))}-\frac12\sigma'(F^{-1}(y))$.

Thus Lamperti converts state-dependent volatility into constant volatility one. Euler through Lamperti uses

$\hat Y_{t_{i+1}}=\hat Y_{t_i}+a(\hat Y_{t_i})\Delta t_{i+1}+\sqrt{\Delta t_{i+1}}Z_{i+1},\qquad \hat X_{t_{i+1}}=F^{-1}(\hat Y_{t_{i+1}})$.

\subsubsection*{GBM transform}

For $dX_t=rX_t\,dt+\sigma X_t\,dW_t$,

$F(x)=\frac{1}{\sigma}\log x,\qquad Y_t=\frac{1}{\sigma}\log X_t,\qquad X_t=e^{\sigma Y_t}$.

Ito's formula gives

$dY_t=\left(\frac{r}{\sigma}-\frac{\sigma}{2}\right)dt+dW_t$.

Euler on $Y$ is exact, so transforming back gives a scheme for $X$ without discretization error.

\subsection{Numerical Examples}

\subsubsection*{Exact grid for deterministic volatility}

For a discretely monitored floating-strike lookback call with payoff $S_T-\min_{i=1,\ldots,m}S_{t_i}$ and

$dS_t=rS_t\,dt+\theta(t)S_t\,dW_t$,

set $Y_t=\log S_t$. Then

$dY_t=(r-\frac12\theta^2(t))dt+\theta(t)dW_t$.

Let $\sigma_i^2=\int_{t_i}^{t_{i+1}}\theta^2(t)\,dt$. The exact grid recursion is

$Y_{i+1}=Y_i+r\Delta t_i-\frac12\sigma_i^2+\sigma_iZ_{i+1},\qquad S_{i+1}=e^{Y_{i+1}}$.

The discounted output is $H=e^{-rT}(S_m-\min(S_1,\ldots,S_m))$. For $\theta(t)=\sqrt{a^2+b^2\sin(2\pi t/T)}$,

$\sigma_i^2=a^2\Delta t_i+\frac{b^2T}{2\pi}\{\cos(2\pi t_i/T)-\cos(2\pi t_{i+1}/T)\}$.

Exact grid simulation removes process discretization error, but a continuously monitored payoff can still be biased because the between-grid minimum is approximated.

\subsubsection*{Artificial-dynamics control}

For $dS_t=rS_tdt+\theta_tS_tdW_t$ and $d\theta_t=a(\Theta-\theta_t)dt+\beta dB_t$, with $\operatorname{corr}(dW,dB)=\rho$, set $Y_t=\log S_t$. Euler gives

$\hat Y_{i+1}=\hat Y_i+(r-\frac12\hat\theta_i^2)\Delta t_i+\hat\theta_i\sqrt{\Delta t_i}Z_{i+1}$,

$\hat\theta_{i+1}=\hat\theta_i+a(\Theta-\hat\theta_i)\Delta t_i+\beta\sqrt{\Delta t_i}R_{i+1}$.

The call output is $X=e^{-rT}(e^{\hat Y_T}-K)^+$. Use the same $Z_i$ to define an artificial GBM log process

$\bar Y_{i+1}=\bar Y_i+(r-\frac12\sigma^2)\Delta t_i+\sigma\sqrt{\Delta t_i}Z_{i+1}$.

Then $V=e^{-rT}(e^{\bar Y_T}-K)^+$ has expectation $\operatorname{BSCall}(S_0,K,T,r,\sigma)$, giving the control output

$X-b\{V-\operatorname{BSCall}(S_0,K,T,r,\sigma)\}$.

Choosing $\sigma$ near the long-run mean $\Theta$ gives strong variance reduction in the lecture example.

\subsubsection*{Heston with cross-entropy IS}

For the Heston model

$dS_t=rS_tdt+\sqrt{\theta_t}S_tdW_t,\qquad d\theta_t=a(b-\theta_t)dt+\sigma\sqrt{\theta_t}dB_t$,

use the nonnegative Euler scheme

$\hat S_{i+1}=\hat S_i+r\hat S_i\Delta t_i+\hat S_i\sqrt{\hat\theta_i\Delta t_i}\,Z_{i+1}$,

$\hat\theta_{i+1}=\max\{0,\hat\theta_i+a(b-\hat\theta_i)\Delta t_i+\sigma\sqrt{\hat\theta_i\Delta t_i}\,R_{i+1}\}$,

where $R_{i+1}=\rho Z_{i+1}+\sqrt{1-\rho^2}Y_{i+1}$. The payoff is $h(X)=e^{-rT}(\hat S_T-K)^+$ for $X=(Z_1,\ldots,Z_m,Y_1,\ldots,Y_m)$.

For IS use $N(\mu,I_{2m})$, $\mu=(\nu,v)$. An initial tilt is

$\nu^0=x(\sqrt{\Delta t_1},\ldots,\sqrt{\Delta t_m}),\qquad v^0=-\frac{\rho}{\sqrt{1-\rho^2}}\nu^0$,

where matching $E[S_T]\approx K$ after replacing $\theta_t$ by $b$ gives

$x=\frac{1}{\sqrt{bT}}\log\left(\frac{K}{S_0}\right)-\frac{r}{\sqrt b}$.

The cross-entropy update is

$\hat\mu^{j+1}=\frac{\sum_{k=1}^Nh(\bar X_k)e^{-\langle\mu^j,\bar X_k\rangle}\bar X_k}{\sum_{k=1}^Nh(\bar X_k)e^{-\langle\mu^j,\bar X_k\rangle}},\qquad \bar X_k\sim N(\mu^j,I_{2m})$.

After $\hat\mu=(\hat\nu,\hat v)$ is chosen, simulate $Z_i\sim N(\hat\nu_i,1)$ and $Y_i\sim N(\hat v_i,1)$ and multiply the payoff by

$\exp\left\{-\sum_{i=1}^m(\hat\nu_iZ_i+\hat v_iY_i)+\frac12\sum_{i=1}^m(\hat\nu_i^2+\hat v_i^2)\right\}$.

The variance reduction is strongest for high strikes.
