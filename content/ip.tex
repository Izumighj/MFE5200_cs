\section{Integer Programming Models}

\subsection{Types of IP}
\subsubsection*{Pure Integer}
All variables are integer-valued.
$x \in \mathbb{Z}^n$

\subsubsection*{Mixed Integer (MIP)}
Some variables are integer, some are continuous.
$x \in \mathbb{Z}^p \times \mathbb{R}^{n-p} $

\subsection{Logical Constraints (with $y \in \{0, 1\}$)}

\subsubsection*{K-out-of-N}
$\sum_{i=1}^N y_i = k$

\subsubsection*{If-Then (Big-M Method)}
If $y=1$, then $f(x) \le b$.
\[ f(x) \le b + M(1-y) \]

\subsubsection*{Prerequisite}
If $y_1=1$, then $y_2=1$.
\[ y_2 \ge y_1 \]
